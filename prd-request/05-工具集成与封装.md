## 2. arXiv 搜索工具（已实现）

当前已在后端实现，需要封装为 LangGraph 可调用的工具函数。

```python
# tools/arxiv_tool.py
import httpx
import xmltodict
import time
from typing import List, Dict

ARXIV_API_URL = "https://export.arxiv.org/api/query"
_last_req_time = 0

def search_arxiv(
    query: str,
    max_results: int = 20,
    sort_by: str = "relevance",
    categories: List[str] = None,
) -> List[Dict]:
    """
    arXiv 论文搜索工具
    
    Args:
        query: 搜索查询（arXiv 语法，如 "CLIP noisy label"）
        max_results: 返回结果数（≤100）
        sort_by: 排序方式 relevance/submittedDate
        categories: 过滤分类，如 ["cs.LG", "cs.CV"]
    
    Returns:
        [{title, summary, authors, categories, published, pdf_link, arxiv_id}]
    """
    global _last_req_time
    
    # 限流 3 秒
    now = time.time()
    since_last = now - _last_req_time
    if since_last < 3:
        time.sleep(3 - since_last)
    
    params = {
        "search_query": query.replace(" ", "+"),
        "max_results": min(max_results, 100),
        "sortBy": sort_by,
        "sortOrder": "descending",
    }
    
    resp = httpx.get(ARXIV_API_URL, params=params, timeout=30)
    resp.raise_for_status()
    _last_req_time = time.time()
    
    data = xmltodict.parse(resp.text)
    entries = data.get("feed", {}).get("entry", [])
    if isinstance(entries, dict):
        entries = [entries]
    
    papers = []
    for entry in entries:
        authors = entry.get("author", [])
        if isinstance(authors, dict):
            authors = [authors]
        
        cats = entry.get("category", [])
        if isinstance(cats, dict):
            cats = [cats]
        cat_terms = [c.get("@term", "") for c in cats]
        
        # 分类过滤
        if categories and not any(c in cat_terms for c in categories):
            continue
        
        links = entry.get("link", [])
        if isinstance(links, dict):
            links = [links]
        
        paper = {
            "arxiv_id": entry.get("id", "").split("/")[-1].split("v")[0],
            "title": entry.get("title", "").strip().replace("\n", " "),
            "summary": entry.get("summary", "").strip().replace("\n", " "),
            "authors": [a.get("name", "") for a in authors[:10]],
            "categories": cat_terms,
            "published": entry.get("published", "")[:10],
            "pdf_link": next(
                (l.get("@href", "") for l in links if "pdf" in l.get("@href", "")),
                "",
            ),
            "arxiv_url": entry.get("id", ""),
        }
        papers.append(paper)
    
    return papers
```

```python
# tools/docx_tool.py
from docx import Document
from docx.shared import Inches, Pt
from docx.enum.text import WD_ALIGN_PARAGRAPH
import os

REPORTS_DIR = "generated_reports"

def create_docx(title: str) -> str:
    """创建新的 Word 文档，返回文件路径"""
    os.makedirs(REPORTS_DIR, exist_ok=True)
    path = os.path.join(REPORTS_DIR, f"{title}.docx")
    
    doc = Document()
    doc.add_heading(title, 0)
    doc.save(path)
    return path

def add_section(filepath: str, heading: str, content: str):
    """在文档中添加新章节"""
    doc = Document(filepath)
    doc.add_heading(heading, 1)
    doc.add_paragraph(content)
    doc.save(filepath)

def add_table(filepath: str, headers: List[str], rows: List[List[str]]):
    """添加对比表格"""
    doc = Document(filepath)
    table = doc.add_table(rows=1 + len(rows), cols=len(headers))
    for i, h in enumerate(headers):
        table.rows[0].cells[i].text = h
    for r, row in enumerate(rows):
        for c, val in enumerate(row):
            table.rows[r + 1].cells[c].text = val
    doc.save(filepath)
```

## 5. 工具注册到 LangGraph

使用 `ToolNode` 注册所有工具：

```python
from langgraph.prebuilt import ToolNode

# 工具函数列表
tools = [
    search_arxiv,
    create_docx,
    add_section,
    add_table,
    create_architecture_diagram,
]

# 工具节点
tool_node = ToolNode(tools)

# 在 Agent 节点中使用
def agent_with_tools(state: AgentState, tools: List) -> AgentState:
    """带工具的 ReAct Agent 节点"""
    # 系统提示包含工具描述
    system_prompt = build_system_prompt_with_tools(tools)
    
    # LLM 调用，允许工具调用
    response = llm_with_tools.invoke([
        {"role": "system", "content": system_prompt},
        *state["messages"],
    ])
    
    # 如果有工具调用请求
    if response.tool_calls:
        for call in response.tool_calls:
            result = execute_tool_call(call, tools)
            state = update_state(state, call, result)
    
    return state
```
